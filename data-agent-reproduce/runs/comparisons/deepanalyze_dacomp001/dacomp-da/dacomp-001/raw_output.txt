```python
import sqlite3
import pandas as pd
import numpy as np

# 连接数据库
conn = sqlite3.connect('dacomp-001.sqlite')

# 读取企业数据
query = """
SELECT 
    e.Enterprise_ID,
    e.Scale,
    e.Credit_Rating,
    e.Revenue,
    e.Profit,
    e.Upstream_Dependency,
    e.Downstream_Dependency,
    ch.Credit_Score,
    ch.Default_Records
FROM enterprises e
JOIN credit_histories ch ON e.Enterprise_ID = ch.Enterprise_ID
"""

df = pd.read_sql_query(query, conn)

# 关闭数据库连接
conn.close()

# 计算风险评分
# 假设风险评分 = (100 - Credit_Score) + Default_Records * 10 + Upstream_Dependency * 5 + Downstream_Dependency * 5
df['Risk_Score'] = (100 - df['Credit_Score']) + df['Default_Records'] * 10 + df['Upstream_Dependency'] * 5 + df['Downstream_Dependency'] * 5

# 根据信用评级和风险评分进行分组
df['Risk_Level'] = pd.cut(df['Risk_Score'], 
                          bins=[0, 30, 60, 90, float('inf')], 
                          labels=['Low', 'Medium-Low', 'Medium-High', 'High'])

# 计算每个风险等级的企业数量
risk_counts = df['Risk_Level'].value_counts()

# 计算每个风险等级的平均额度和利率
# 假设额度和利率与风险等级相关
# 低风险：额度高，利率低
# 高风险：额度低，利率高

# 假设总信用额度为100000000元
total_credit = 100000000

# 定义分配规则
# 风险等级分配比例
risk_allocation = {
    'Low': 0.4,      # 40%
    'Medium-Low': 0.3, # 30%
    'Medium-High': 0.2, # 20%
    'High': 0.1      # 10%
}

# 计算每个风险等级的额度
risk_limits = {}
for level, ratio in risk_allocation.items():
    risk_limits[level] = total_credit * ratio

# 利率设定（假设）
# 风险等级对应的利率
risk_rates = {
    'Low': 0.04,     # 4%
    'Medium-Low': 0.06, # 6%
    'Medium-High': 0.08, # 8%
    'High': 0.12     # 12%
}

# 输出结果
print("企业风险等级分布:")
print(risk_counts)
print("\n额度分配:")
for level, limit in risk_limits.items():
    print(f"{level}: {limit:.0f} 元")

print("\n利率分配:")
for level, rate in risk_rates.items():
    print(f"{level}: {rate*100:.0f}%")

# 保存结果到CSV文件
df.to_csv('enterprise_risk_analysis.csv', index=False)
```

```python
import sqlite3
import pandas as pd

conn = sqlite3.connect('dacomp-001.sqlite')
```

```python
query = """
SELECT 
    `Enterprise Code`,
    `Credit Rating`,
    `Defaulted`
FROM ch___company_info
"""

df = pd.read_sql_query(query, conn)

# 关闭数据库连接
```
